### 4.1 认证管理（Background）

#### 4.1.1 登录流程

```javascript
// background/auth.js

// 登录
async function login(username, password) {
  const apiUrl = await getApiUrl();
  
  const response = await fetch(`${apiUrl}/auth/login`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ username, password })
  });
  
  const data = await response.json();
  
  if (data.success) {
    // 存储 Token 和用户信息
    await chrome.storage.local.set({
      accessToken: data.token,
      user: data.user,
      loginTime: Date.now()
    });
    
    return { success: true, user: data.user };
  } else {
    throw new Error(data.message || '登录失败');
  }
}

// 检查 Token 是否有效
async function isTokenValid() {
  const { accessToken, loginTime } = await chrome.storage.local.get([
    'accessToken',
    'loginTime'
  ]);
  
  if (!accessToken) return false;
  
  // 检查是否过期（7天 - 提前1天刷新）
  const tokenExpireTime = loginTime + (6 * 24 * 60 * 60 * 1000);
  
  return Date.now() < tokenExpireTime;
}

// 获取有效 Token（自动刷新）
async function getValidToken() {
  if (await isTokenValid()) {
    const { accessToken } = await chrome.storage.local.get(['accessToken']);
    return accessToken;
  } else {
    // Token 过期，需要重新登录
    throw new Error('Token 已过期，请重新登录');
  }
}
```

### 4.2 数据解析（Content Script）

#### 4.2.1 搜索页解析

```javascript
// content/pages/SearchPage.js

class SearchPage {
  constructor() {
    this.observer = null;
  }
  
  // 初始化
  init() {
    // 监听页面变化（TikTok 是 SPA）
    this.observePageChanges();
    
    // 解析当前页面
    this.parseSearchResults();
  }
  
  // 监听页面变化
  observePageChanges() {
    this.observer = new MutationObserver(() => {
      this.parseSearchResults();
    });
    
    this.observer.observe(document.body, {
      childList: true,
      subtree: true
    });
  }
  
  // 解析搜索结果
  parseSearchResults() {
    // 视频卡片选择器（需要根据 TikTok 实际 DOM 调整）
    const videoCards = document.querySelectorAll('[data-e2e="video-card"]');
    
    videoCards.forEach(card => {
      const videoData = this.parseVideoCard(card);
      
      // 显示覆盖层
      this.showOverlay(card, videoData);
    });
  }
  
  // 解析视频卡片
  parseVideoCard(card) {
    return {
      videoId: card.querySelector('a')?.href?.split('/')?.pop(),
      author: {
        username: card.querySelector('[data-e2e="video-author-uniqueid"]')?.textContent,
        nickname: card.querySelector('[data-e2e="video-author-nickname"]')?.textContent
      },
      stats: {
        plays: card.querySelector('[data-e2e="video-plays"]')?.textContent,
        likes: card.querySelector('[data-e2e="video-likes"]')?.textContent
      }
    };
  }
  
  // 显示覆盖层
  showOverlay(card, videoData) {
    // 创建覆盖层元素
    const overlay = document.createElement('div');
    overlay.className = 'lf-overlay';
    overlay.innerHTML = `
      <div class="lf-metrics">
        <span class="lf-tag">GMV: $1,250</span>
        <span class="lf-tag">FV: 0.08</span>
      </div>
      <button class="lf-import-btn">导入到 LazyFirst</button>
    `;
    
    card.appendChild(overlay);
    
    // 绑定导入按钮事件
    overlay.querySelector('.lf-import-btn').addEventListener('click', () => {
      this.importToLazyFirst(videoData.author);
    });
  }
  
  // 导入到 LazyFirst
  async importToLazyFirst(author) {
    // 发送消息到 Background
    chrome.runtime.sendMessage({
      type: 'IMPORT_INFLUENCER',
      data: author
    }, response => {
      if (response.success) {
        alert('导入成功！');
      } else {
        alert('导入失败：' + response.message);
      }
    });
  }
}
```

```javascript
// content/pages/ProfilePage.js

class ProfilePage {
  constructor() {
    this.username = window.location.pathname.split('/')[1]?.replace('@', '');
  }
  
  // 初始化
  init() {
    // 解析达人信息
    const influencerData = this.parseProfile();
    
    // 查询 LazyFirst 中的历史数据
    this.fetchHistoricalData(influencerData);
    
    // 显示覆盖层
    this.showProfileOverlay(influencerData);
  }
  
  // 解析达人主页
  parseProfile() {
    return {
      tiktokId: '@' + this.username,
      tiktokName: document.querySelector('[data-e2e="user-subtitle"]')?.textContent,
      followerCount: this.parseNumber(
        document.querySelector('[data-e2e="followers-count"]')?.textContent
      ),
      followingCount: this.parseNumber(
        document.querySelector('[data-e2e="following-count"]')?.textContent
      ),
      bio: document.querySelector('[data-e2e="user-bio"]')?.textContent,
      avatar: document.querySelector('[data-e2e="user-avatar"]')?.src
    };
  }
  
  // 查询历史数据
  async fetchHistoricalData(influencerData) {
    // 发送消息到 Background
    chrome.runtime.sendMessage({
      type: 'FETCH_INFLUENCER',
      tiktokId: influencerData.tiktokId
    }, response => {
      if (response.success && response.influencer) {
        // 显示历史数据
        this.showHistoricalData(response.influencer);
      }
    });
  }
  
  // 显示历史数据
  showHistoricalData(influencer) {
    const overlay = document.createElement('div');
    overlay.className = 'lf-profile-overlay';
    overlay.innerHTML = `
      <h3>LazyFirst 数据</h3>
      <p>最新粉丝数：${influencer.latestFollowers}</p>
      <p>最新 GMV：${influencer.latestGmv}</p>
      <p>月销件数：${influencer.monthlySalesCount}</p>
      <p>归属 BD：${influencer.latestMaintainerName}</p>
    `;
    
    // 插入到页面合适位置
    document.querySelector('[data-e2e="user-info"]').appendChild(overlay);
  }
  
  // 解析数字（处理 "1.2M" 格式）
  parseNumber(text) {
    if (!text) return 0;
    
    const match = text.match(/([\d.]+)([KMB])?/i);
    if (!match) return parseInt(text) || 0;
    
    const num = parseFloat(match[1]);
    const unit = match[2]?.toUpperCase();
    
    switch (unit) {
      case 'K': return num * 1000;
      case 'M': return num * 1000000;
      case 'B': return num * 1000000000;
      default: return num;
    }
  }
}
```

### 4.3 数据同步（Background）

#### 4.3.1 主逻辑

```javascript
// background/sync.js

// 导入达人（核心逻辑）
async function importInfluencer(authorData) {
  try {
    const apiUrl = await getApiUrl();
    const token = await getValidToken();
    
    // 1. 检查达人是否存在
    const existingInfluencer = await checkInfluencerExists(authorData.tiktokId);
    
    if (existingInfluencer) {
      // 2a. 达人已存在 → 添加维护记录
      await addMaintenanceRecord(existingInfluencer._id, authorData);
      
      return {
        success: true,
        message: '已更新维护记录',
        influencerId: existingInfluencer._id
      };
    } else {
      // 2b. 达人不存在 → 创建新达人
      const newInfluencer = await createInfluencer(authorData);
      
      // 添加维护记录
      await addMaintenanceRecord(newInfluencer._id, authorData);
      
      return {
        success: true,
        message: '已创建达人并添加维护记录',
        influencerId: newInfluencer._id
      };
    }
  } catch (error) {
    return {
      success: false,
      message: error.message
    };
  }
}

// 检查达人是否存在
async function checkInfluencerExists(tiktokId) {
  const apiUrl = await getApiUrl();
  const token = await getValidToken();
  
  const response = await fetch(
    `${apiUrl}/influencers?tiktokId=${encodeURIComponent(tiktokId)}`,
    {
      headers: {
        'Authorization': `Bearer ${token}`
      }
    }
  );
  
  const data = await response.json();
  
  if (data.success && data.data.length > 0) {
    return data.data[0];
  } else {
    return null;
  }
}

// 创建新达人
async function createInfluencer(authorData) {
  const apiUrl = await getApiUrl();
  const token = await getValidToken();
  const { user } = await chrome.storage.local.get(['user']);
  
  const response = await fetch(`${apiUrl}/influencers`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      tiktokName: authorData.tiktokName,
      tiktokId: authorData.tiktokId,
      latestFollowers: authorData.followerCount,
      poolType: 'private',
      assignedTo: user._id
    })
  });
  
  const data = await response.json();
  
  if (data.success) {
    return data.data;
  } else {
    throw new Error(data.message || '创建达人失败');
  }
}

// 添加维护记录
async function addMaintenanceRecord(influencerId, authorData) {
  const apiUrl = await getApiUrl();
  const token = await getValidToken();
  const { user } = await chrome.storage.local.get(['user']);
  
  const response = await fetch(`${apiUrl}/influencer-maintenances`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      influencerId: influencerId,
      followers: authorData.followerCount,
      remark: 'Chrome 插件自动更新',
      maintainerId: user._id,
      maintainerName: user.realName
    })
  });
  
  const data = await response.json();
  
  if (!data.success) {
    throw new Error(data.message || '添加维护记录失败');
  }
}
```

#### 4.3.2 Background Service Worker 消息处理

```javascript
// background/background.js

chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
  if (request.type === 'IMPORT_INFLUENCER') {
    // 导入达人
    importInfluencer(request.data)
      .then(result => sendResponse(result))
      .catch(error => sendResponse({ success: false, message: error.message }));
    
    return true; // 异步响应
  }
  
  if (request.type === 'FETCH_INFLUENCER') {
    // 查询达人
    checkInfluencerExists(request.tiktokId)
      .then(influencer => sendResponse({ success: true, influencer }))
      .catch(error => sendResponse({ success: false, message: error.message }));
    
    return true;
  }
});
```

### 5.2 CORS 配置（需要修改）

 Chrome Extension 需要跨域调用 API，需要在后端添加 CORS 支持：

```javascript
// server/middleware/cors.js（新增）

const cors = (req, res, next) => {
  // 允许 Chrome Extension 访问
  const allowedOrigins = [
    'chrome-extension://xxxxx',  // 插件 ID（需要动态获取）
    'http://localhost:5175',    // 开发环境
    'https://lazyfirst.com'     // 生产环境
  ];
  
  const origin = req.headers.origin;
  
  if (allowedOrigins.includes(origin) || origin?.startsWith('chrome-extension://')) {
    res.header('Access-Control-Allow-Origin', origin);
    res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE');
    res.header('Access-Control-Allow-Headers', 'Authorization, Content-Type');
    res.header('Access-Control-Allow-Credentials', 'true');
  }
  
  if (req.method === 'OPTIONS') {
    return res.sendStatus(200);
  }
  
  next();
};

module.exports = cors;
```

## 八、关键技术难点

### 8.1 TikTok DOM 解析

**问题**：TikTok 页面是动态加载的，DOM 结构可能变化

**解决方案**：
1. 使用 `MutationObserver` 监听页面变化
2. 定期维护选择器（根据 TikTok 页面更新）
3. 优先使用 `data-e2e` 属性（TikTok 专用）

### 8.2 Chrome Extension 跨域请求

**问题**：Content Script 不能直接跨域请求

**解决方案**：
1. 使用 Background Service Worker 转发请求
2. 配置 CORS 允许 Chrome Extension 访问

### 8.3 Token 管理

**问题**：JWT Token 会过期

**解决方案**：
1. 存储登录时间和 Token
2. 每次请求前检查是否过期
3. 提前刷新 Token（如果实现了 Refresh Token）
